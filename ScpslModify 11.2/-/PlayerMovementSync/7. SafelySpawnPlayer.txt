{

	this._successfullySpawned = true;

{
	if (thisProcessId != this._spawnProcessId)
	{
		yield break;

	Action<global::ReferenceHub> onPlayerSpawned = global::PlayerMovementSync.OnPlayerSpawned;
	if (onPlayerSpawned != null)

		onPlayerSpawned(this._hub);
