int num = -1;
if (armor != null)
{
	if (ammoType != ItemType.Ammo12gauge && ammoType != ItemType.Ammo44cal)
	{
		switch (armor.ItemTypeId)
		{
			case ItemType.ArmorLight:
				num = 200;
				break;
			case ItemType.ArmorCombat:
				num = 300;
				break;
			case ItemType.ArmorHeavy:
				num = 400;
				break;
		}
	}
	else
	{
		switch (armor.ItemTypeId)
		{
			case ItemType.ArmorLight:
				num = 40;
				break;
			case ItemType.ArmorCombat:
				num = 70;
				break;
			case ItemType.ArmorHeavy:
				num = 100;
				break;
		}
	}
}
else if (ammoType == ItemType.Ammo12gauge || ammoType == ItemType.Ammo44cal)
{
	num = 20;
}
else
{
	num = 100;
}
return (ushort)num;
